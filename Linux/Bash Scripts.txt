#!/bin/bash

mkdir ~/backups/{freemem,diskuse,openlist,freedisk}


cp free -h > ~/backups/freemem/free_mem.txt
cp used -h > ~/backups/diskuse/disk_usage.txt
cp lsof /dev/null -h  > ~/backups/open_list.txt
cp df -h > ~/backups/freedisk/free_disk.txt

------------------------------------------

if [ -d ~$HOME/research] 
then
echo 'research directory exists'
else
mkdir $HOME/research
fi
